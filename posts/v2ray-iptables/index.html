v2ray使用iptables实现动态端口

四月 8, 2023

v2ray支持动态端口（dynamicPort），但这是协议的一种，无法与ws+tls共存，今天看到别人分享的用iptables去实现动态端口觉得思路很好 于是记录一下

比如当前的v2ray端口是88
那么使用iptables去开一个端口范围转发到88

iptables -t nat -A PREROUTING -p tcp --dport 40000:60000 -j REDIRECT --to-ports 88

上面的例子是将端口4000-6000的范围全部转发到88
可以将json格式的配置文件放置在国内环境下，使用PHP去监控并自动修改

<?php
// 读取 us.json 文件
$json = file_get_contents('us.json');
$data = json_decode($json, true);

// 获取 address 和 port
$address = $data['outbounds'][0]['settings']['vnext'][0]['address'];
$port = $data['outbounds'][0]['settings']['vnext'][0]['port'];

// 判断 TCP 连通性
function test_tcp_connection($ip, $port) {
    $timeout = 3;
    $socket = @fsockopen($ip, $port, $errno, $errstr, $timeout);
    if (!$socket) {
        return false; // connection failed
    }
    fclose($socket);
    return true; // connection successful
}

if (!test_tcp_connection($address,$port) && $port >= 40000 && $port <= 60000) {
    // 如果不能连接且端口在 40000-60000 范围内，则循环判断
    $port++;
    while ($port <= 60000) {
        if (test_tcp_connection($address,$port)) {
            break;
        }
        $port++;
    }
} elseif (!test_tcp_connection($address,$port) && $port < 40000) {
    // 如果不能连接且端口小于 40000，则从 40000 开始
    $port = 40000;
    while ($port <= 60000) {
        if (test_tcp_connection($address,$port)) {
            break;
        }
        $port++;
    }
}

if (test_tcp_connection($address,$port)) {
    echo "TCP 连通，端口号为：" . $port . "<br>";

    // 如果原来的端口已经连通，则不更改
    if ($port === $data['outbounds'][0]['settings']['vnext'][0]['port']) {
        echo "原来的端口已经连通，不需要更改。";
    } else {
        // 更新 us.json 文件中的 port 字段
        $data['outbounds'][0]['settings']['vnext'][0]['port'] = $port;
        $updatedJson = json_encode($data, JSON_UNESCAPED_UNICODE | JSON_PRETTY_PRINT);
        file_put_contents('us.json', $updatedJson);

        echo "端口已经更改为：" . $port;
    }
} else {
    echo "TCP 连通失败。";
}
?>
